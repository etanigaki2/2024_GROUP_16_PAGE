229void MainWindow::updateRender()
230{
231 renderer->RemoveAllViewProps();
232 int topLevelCount = partList->rowCount(QModelIndex());
233
234 for (int i = 0; i < topLevelCount; ++i) {
235 QModelIndex topIndex = partList->index(i, 0, QModelIndex());
236 updateRenderFromTree(topIndex);
237 }
238
239 if (renderer->GetActors()->GetNumberOfItems() > 0) {
240 renderer->ResetCamera();
241 }
242
243 renderWindow->Render();
244}

250void MainWindow::updateRenderFromTree(const QModelIndex& index)
251{
252 if (!index.isValid()) return;
253
254 ModelPart* selectedPart = static_cast<ModelPart*>(index.internalPointer());
255
256 if (selectedPart && selectedPart->visible()) {
257 vtkSmartPointer<vtkActor> actor = selectedPart->getActor();
258 if (actor) {
259 renderer->AddActor(actor);
260 }
261 }
262
263 int rows = partList->rowCount(index);
264 for (int i = 0; i < rows; i++) {
265 updateRenderFromTree(partList->index(i, 0, index));
266 }
267}
